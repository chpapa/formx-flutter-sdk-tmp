Widget preview({
  Color detectedColor = Colors.blue,
  Color detectingColor = Colors.green,
  Color undetectedColor = Colors.red,
}) {
  const codec = StandardMessageCodec();

  final parameters = {
    'viewId': viewId,
    'detectedColor': detectedColor.value,
    'detectingColor': detectingColor.value,
    'undetectedColor': undetectedColor.value,
  };

  if (viewId != kUninitializedId) {
    return Container(
      clipBehavior: Clip.hardEdge,
      decoration: const BoxDecoration(),
      child: switch (defaultTargetPlatform) {
        TargetPlatform.android => AndroidView(
            viewType: viewType,
            creationParams: parameters,
            creationParamsCodec: codec,
          ),
        TargetPlatform.iOS => UiKitView(
            viewType: viewType,
            creationParams: parameters,
            creationParamsCodec: codec,
          ),
        _ => throw Exception('Unsupported platform'),
      },
    );
  }

  return Container();
}
